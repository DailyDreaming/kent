    # no AGP, make one from the fasta file, the sed changes type D to type W
    # the seeming blanks in the sed are tabs via ctrl-V I
    hgFakeAgp -minContigGap=1 \
/hive/data/genomes/caePb3/ws220/sequences/dna/c_brenneri.WS220.dna.fa.gz \
	stdout | sed -e "s/	D	/	W	/" \
	> caePb3.ws220.fake.agp

    faSize sequences/dna/c_brenneri.WS220.dna.fa.gz

    #	Loaded 1354976 elements of size 3

    #	170097428 bases of 170097428 (100.000%) in intersection

	-bed=stdout | gzip -c > cleanWMask.bed.gz
